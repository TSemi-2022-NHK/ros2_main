cmake_minimum_required(VERSION 3.8)
project(bno055_io)

if(CMAKE_COMPILER_IS_GNUCXX OR CMAKE_CXX_COMPILER_ID MATCHES "Clang")
  add_compile_options(-Wall -Wextra -Wpedantic)
endif()

# find dependencies
find_package(ament_cmake REQUIRED)
find_package(ament_cmake_ros REQUIRED)
find_package(rclcpp REQUIRED)
find_package(sensor_msgs REQUIRED)
find_package(socketcan_interface REQUIRED)
find_package(utilities REQUIRED)

add_library(bno055_io_node src/bno055_io_node.cpp)
target_compile_features(bno055_io_node PUBLIC c_std_99 cxx_std_17)  # Require C99 and C++17
target_include_directories(bno055_io_node PUBLIC
  $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
  $<INSTALL_INTERFACE:include>)
ament_target_dependencies(
  bno055_io_node
  "rclcpp"
  "sensor_msgs"
  "socketcan_interface"
        "utilities"
)

# Causes the visibility macros to use dllexport rather than dllimport,
# which is appropriate when building the dll but not consuming it.
target_compile_definitions(bno055_io_node PRIVATE "BNO055_IO_BUILDING_LIBRARY")

ament_export_targets(export_${PROJECT_NAME} HAS_LIBRARY_TARGET)
ament_export_dependencies(
        rclcpp
        sensor_msgs
        socketcan_interface
        utilities
)

install(
  DIRECTORY include/
  DESTINATION include
)
install(
  TARGETS bno055_io_node
  EXPORT export_${PROJECT_NAME}
  ARCHIVE DESTINATION lib
  LIBRARY DESTINATION lib
  RUNTIME DESTINATION bin
)

if(BUILD_TESTING)
  find_package(ament_lint_auto REQUIRED)
  # the following line skips the linter which checks for copyrights
  # uncomment the line when a copyright and license is not present in all source files
  #set(ament_cmake_copyright_FOUND TRUE)
  # the following line skips cpplint (only works in a git repo)
  # uncomment the line when this package is not in a git repo
  #set(ament_cmake_cpplint_FOUND TRUE)
  ament_lint_auto_find_test_dependencies()
endif()

ament_export_include_directories(
  include
)
ament_export_libraries(
  bno055_io_node
)
ament_export_targets(
  export_${PROJECT_NAME}
)

ament_package()
